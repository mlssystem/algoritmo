var

i, j, aux: Inteiro

inicio

Para i <- 1 ate 9 faca
Para j <- i+1 ate 10 faca
Se (v[i] > v[j]) entao
aux <- v[i]
v[i] <- v[j]
v[j] <- aux
FimSe
